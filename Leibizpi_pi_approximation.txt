function Leibizpi($n){
	$Pi_approximation = "";
    $Pole = 1;
	for($i=1; $i<=$n; $i+=2){
    $Pi_approximation += $Pole*(1/$i);
	$Pole *= -1;
    }
    $Pi_approximation = round($Pi_approximation,3);
    return $Pi_approximation;
}
